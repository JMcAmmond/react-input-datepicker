## Install

```bash
npm install --save react-input-datepicker
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-input-datepicker'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
